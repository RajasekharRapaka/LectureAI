# LectureAI — Environment Variables
# Copy this to .env:  cp .env.example .env

# ── Required ────────────────────────────────────────────────────────
# Get a free key at: https://aistudio.google.com/apikey
GEMINI_API_KEY=your_gemini_api_key_here

# ── Model (optional) ────────────────────────────────────────────────
# gemini-2.5-pro   = best quality, slower  (default)
# gemini-2.5-flash = fast + great quality
# gemini-2.0-flash = fastest
GEMINI_MODEL=gemini-2.5-pro

# ── YouTube Cookies (REQUIRED to fix transcript fetch errors) ───────
#
# YouTube blocks automated requests without real browser cookies.
# If you see "Could not fetch transcript", do this:
#
#  1. Install "Get cookies.txt LOCALLY" Chrome extension:
#     https://chromewebstore.google.com/detail/cclelndahbckbenkjhflpdbgdldlbecc
#  2. Go to https://www.youtube.com and log in
#  3. Click the extension → Export → save as cookies.txt
#  4. Set the path here (use your actual username):
#
YOUTUBE_COOKIES_FILE=/Users/YOUR_MAC_USERNAME/cookies.txt

# On EC2: YOUTUBE_COOKIES_FILE=/home/ubuntu/cookies.txt
# Tip: Re-export every 2-4 weeks as cookies expire.
